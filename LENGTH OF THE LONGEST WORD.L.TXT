%{
#include <stdio.h>
#include <string.h>
int max_length = 0;
%}

%%
[a-zA-Z]+ {
    int length = strlen(yytext);
    if (length > max_length) {
        max_length = length;
    }
}
.|\n       /* ignore other characters */
%%

int main() {
    yylex(); // Start the lexer
    printf("Length of the longest word: %d\n", max_length);
    return 0;
}
